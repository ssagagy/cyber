T1036.008.01 Masquerade File Type

D3FEND

MITRE ATT&CK 액션을 기준으로 대응 방안을 작성

Detection

  • Action = "File-Rename" AND
  • TargetFileFormat = "EXE" OR "DLL" AND
  • TargetFileExetentionBefore != TargetFileExetentionAfter AND
  • TargetFileExetention_Before = "pdf" OR "hwp" OR "hwpx" OR "doc" OR "docx" OR "dotm" OR "dotx" OR "docm" OR "ppt" OR "pptx" OR "ppam" OR "xls" OR "xlsx" OR "xlsb" OR "xlsm" OR "xlam" OR "csv" OR "rtf" OR "ost" OR "jpg" OR "jpeg" OR "jpe" OR "gif" OR "png" OR "bmp" OR "txt" OR "xml" OR "zip" AND
  • TargetFileExetention_After = "exe" OR "dll"

    Detection(EDR)
